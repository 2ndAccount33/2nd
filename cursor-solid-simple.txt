JSX (SetuLandingPage.jsx):
```jsx
useEffect(() => {
  const cursor = document.createElement('div');
  cursor.className = 'custom-cursor';
  document.body.appendChild(cursor);
  
  const move = (e) => {
    cursor.style.left = e.clientX + 'px';
    cursor.style.top = e.clientY + 'px';
  };
  
  const grow = () => cursor.classList.add('hover');
  const shrink = () => cursor.classList.remove('hover');
  const clickStart = () => cursor.classList.add('click');
  const clickEnd = () => cursor.classList.remove('click');
  
  window.addEventListener('mousemove', move);
  window.addEventListener('mousedown', clickStart);
  window.addEventListener('mouseup', clickEnd);
  
  document.querySelectorAll('a, button').forEach(el => {
    el.addEventListener('mouseenter', grow);
    el.addEventListener('mouseleave', shrink);
  });
  
  return () => {
    window.removeEventListener('mousemove', move);
    window.removeEventListener('mousedown', clickStart);
    window.removeEventListener('mouseup', clickEnd);
    cursor.remove();
  };
}, []);
```
